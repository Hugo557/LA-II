            escribir("Revisando promedio...");
            repite {
                escribir("Calculando promedio...");
            } mientras (%prom < 8);
            regresar;
        }

        switch (&edad) {
        }

        escribir("Proceso finalizado");
        regresar;
    }

    // Segundo método
